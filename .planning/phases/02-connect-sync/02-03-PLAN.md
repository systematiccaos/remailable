       IMPORTANT IMPLEMENTATION NOTES for CXX-Qt 0.8:
       - The `extern "RustQt"` block defines QML-visible types
       - `#[qproperty]` makes a field accessible from QML with automatic change notifications
       - `#[qinvokable]` makes a method callable from QML
       - For the global Storage: Use `lazy_static` or `once_cell` to initialize once. The Storage is created in `main()` and stored in a `static mut` or `Mutex<Storage>` that the QML-facing methods can access.
       - Actually, better approach: Use `Rc<RefCell<Storage>>` shared between AppModel and AccountListModel. Pass it during construction in main().
       - For account list data: QML ListView needs roles. CXX-Qt 0.8 can use `QAbstractListModel` — but this requires careful implementation. For v1, use the `get_account_*` invokable methods that QML calls by index, which is simpler than a full list model.

       pub mod cxxqt;
       pub mod account;
       pub mod storage;
       pub mod imap_conn;
       pub mod smtp_conn;
       pub mod sync;

       use cxx_qt::casting::Upcast;
       use cxx_qt_lib::{QGuiApplication, QQmlApplicationEngine, QQmlEngine, QUrl};
       use std::sync::Mutex;

       // Global storage accessible from CXX-Qt bridge
       pub static STORAGE: once_cell::sync::Lazy<Mutex<storage::Storage>> =
           once_cell::sync::Lazy::new(|| {
               let db_path = dirs::data_dir()
                   .unwrap_or_else(|| std::path::PathBuf::from("."))
                   .join("remailable")
                   .join("remailable.db");
               if let Some(parent) = db_path.parent() {
                   std::fs::create_dir_all(parent).ok();
               }
               let db = storage::Storage::open(db_path.to_str().unwrap_or("remailable.db"))
                   .expect("Failed to open database");
               Mutex::new(db)
           });

       fn main() {
           // Initialize storage (accessing Lazy static triggers initialization)
           let _ = STORAGE.len(); // just to ensure initialization

           let mut app = QGuiApplication::new();
           let mut engine = QQmlApplicationEngine::new();

           if let Some(engine) = engine.as_mut() {
               engine.load(&QUrl::from("qrc:/qt/qml/io/remailable/Remailable/qml/main.qml"));
           }

           if let Some(engine) = engine.as_mut() {
               let engine: core::pin::Pin<&mut QQmlEngine> = engine.upcast_pin();
               engine.on_quit(|_| { println!("remailable: QML engine quit"); }).release();
           }

           if let Some(app) = app.as_mut() { app.exec(); }
       }
       ```

       pub struct AppModelRust {
           current_view: QString,
           active_account_id: QString,
           sync_status_text: QString,

           validation_error: QString,

                   ..Default::default()

           }
       }
       ```
